## 🚀 RAG 实现方案

### 整体架构
```
用户提问 → 向量检索知识库 → 获取相关知识 → 结合知识+题目 → Qwen3-8B 推理 → 输出答案 [A-D]
```

#### 步骤 2: 构建向量知识库
```python
# rag_builder.py
from langchain.document_loaders import PyPDFLoader, TextLoader
from langchain.text_splitter import RecursiveCharacterTextSplitter
from langchain.vectorstores import FAISS
from langchain.embeddings import HuggingFaceEmbeddings
import os

def build_knowledge_base(docs_folder, output_path):
    """构建航运知识向量库"""
    # 1. 加载文档
    documents = []
    for file in os.listdir(docs_folder):
        if file.endswith('.pdf'):
            loader = PyPDFLoader(os.path.join(docs_folder, file))
            documents.extend(loader.load())
        elif file.endswith('.txt'):
            loader = TextLoader(os.path.join(docs_folder, file))
            documents.extend(loader.load())

    # 2. 文本分块
    text_splitter = RecursiveCharacterTextSplitter(
        chunk_size=500,
        chunk_overlap=50
    )
    chunks = text_splitter.split_documents(documents)

    # 3. 生成向量并保存
    embeddings = HuggingFaceEmbeddings(
        model_name='BAAI/bge-small-zh-v1.5'
    )
    vectorstore = FAISS.from_documents(chunks, embeddings)
    vectorstore.save_local(output_path)

    return vectorstore

# 使用示例
if __name__ == '__main__':
    vectorstore = build_knowledge_base('knowledge_docs/', 'vectorstore/')
    print(f"知识库构建完成，共 {vectorstore.index.ntotal} 个向量")
```

### 3. RAG 增强推理引擎
```python
# rag_engine.py
import re
from openai import OpenAI
from langchain.vectorstores import FAISS
from langchain.embeddings import HuggingFaceEmbeddings

class RAGMaritimeQA:
    def __init__(self, api_key, base_url, vectorstore_path):
        self.client = OpenAI(api_key=api_key, base_url=base_url)

        # 加载向量知识库
        embeddings = HuggingFaceEmbeddings(model_name='BAAI/bge-small-zh-v1.5')
        self.vectorstore = FAISS.load_local(vectorstore_path, embeddings)

        self.system_prompt = """你是航运领域专家。请根据提供的参考资料回答单选题。
要求：
1. 只输出选项字母（A/B/C/D）
2. 必须用方括号包裹，如 [A]
3. 不要输出任何解释"""

    def retrieve_knowledge(self, question, k=3):
        """检索相关知识"""
        docs = self.vectorstore.similarity_search(question, k=k)
        context = "\n\n".join([doc.page_content for doc in docs])
        return context

    def format_prompt(self, question, options, context):
        """格式化 RAG prompt"""
        return f"""参考资料：
{context}

请根据以上参考资料回答单选题：

题目：{question}
选项：
A. {options['A']}
B. {options['B']}
C. {options['C']}
D. {options['D']}

答："""

    def extract_answer(self, response):
        """提取答案"""
        match = re.search(r'\[([A-D])\]', response)
        return match.group(1) if match else 'A'

    def answer_question(self, question_data):
        """RAG 增强问答"""
        # 1. 检索相关知识
        context = self.retrieve_knowledge(question_data['question'])

        # 2. 构建 prompt
        prompt = self.format_prompt(
            question_data['question'],
            question_data['options'],
            context
        )

        # 3. 调用 Qwen3-8B
        response = self.client.chat.completions.create(
            model="qwen3-8b",
            messages=[
                {'role': 'system', 'content': self.system_prompt},
                {'role': 'user', 'content': prompt}
            ],
            extra_body={"enable_thinking": False}
        )

        # 4. 提取答案
        answer = self.extract_answer(response.choices[0].message.content)
        return f'[{answer}]'

    def batch_inference(self, questions, output_file):
        """批量推理"""
        import json
        results = []

        for idx, q in enumerate(questions):
            print(f"处理题目 {idx+1}/{len(questions)}")
            prediction = self.answer_question(q)
            results.append({
                'index': idx,
                'input': q['question'],
                'prediction': prediction
            })

        # 保存结果
        with open(output_file, 'w', encoding='utf-8') as f:
            for r in results:
                f.write(json.dumps(r, ensure_ascii=False) + '\n')

        print(f"推理完成，结果已保存到 {output_file}")
```

### 4. 网页应用集成
修改 [app.py](app.py) 以集成 RAG 功能：

```python
# app.py (简化版)
from flask import Flask, render_template, request, jsonify
from rag_engine import RAGMaritimeQA
import os

app = Flask(__name__)

# 初始化 RAG 引擎
rag_engine = RAGMaritimeQA(
    api_key=os.getenv("DASHSCOPE_API_KEY", "your-api-key"),
    base_url="https://dashscope.aliyuncs.com/compatible-mode/v1",
    vectorstore_path="vectorstore/"
)

@app.route('/')
def index():
    return render_template('index.html')

@app.route('/chat', methods=['POST'])
def chat():
    data = request.json
    user_message = data.get('message', '').strip()

    if not user_message:
        return jsonify({'error': '消息不能为空'}), 400

    try:
        # 使用 RAG 回答
        question_data = {
            'question': user_message,
            'options': {'A': '', 'B': '', 'C': '', 'D': ''}  # 简化处理
        }
        answer = rag_engine.answer_question(question_data)
        return jsonify({'reply': answer})
    except Exception as e:
        return jsonify({'error': str(e)}), 500

if __name__ == '__main__':
    print("启动 MaritimeBench RAG 系统...")
    print("访问 http://127.0.0.1:5000")
    app.run(host='127.0.0.1', port=5000, debug=False)
```
